1. Получить последние посты конкретного пользователя
	SELECT p.id, p.title, p.body
	FROM posts p
	WHERE p.user_id = 42
	ORDER BY p.id DESC
	LIMIT 10;

2. Подсчитать количество постов у каждого пользователя
	SELECT u.id, u.username, COUNT(p.id) AS posts_count
	FROM users u
	LEFT JOIN posts p ON u.id = p.user_id
	GROUP BY u.id, u.username
	ORDER BY posts_count DESC
	LIMIT 20;

3. Найти пользователей без постов
	SELECT u.id, u.username
	FROM users u
	LEFT JOIN posts p ON u.id = p.user_id
	WHERE p.id IS NULL;

4. Поиск постов по ключевому слову в заголовке
	SELECT id, title, body
	FROM posts
	WHERE title ILIKE '%postgres%';

5. Последние посты всех пользователей с лимитом на каждого
	SELECT DISTINCT ON (p.user_id) p.user_id, p.id, p.title
	FROM posts p
	ORDER BY p.user_id, p.id DESC;

6. Среднее количество символов в постах пользователя
	SELECT u.id, u.username, AVG(LENGTH(p.body)) AS avg_length
	FROM users u
	JOIN posts p ON u.id = p.user_id
	GROUP BY u.id, u.username;
